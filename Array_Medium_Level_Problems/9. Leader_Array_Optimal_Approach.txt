#include<iostream>
#include<vector>
using namespace std;

void leader_array(int arr[], int n)
{
  int max_val;
  vector<int>result;
  max_val = arr[n-1];
  result.push_back(max_val);
  for(int i=n-2; i>=0; i--)
  {
    if(arr[i] < max_val)
    {
      max_val = max(max_val, arr[i-1]);
      result.push_back(arr[i]);
    }
    else
    {
      max_val = max(max_val,arr[i]);
    }
  }
  for(int i=0; i<result.size(); i++)
  {
    cout<<result[i]<<" ";
  }
}

int main()
{
  int arr[] = {4,7,9,5,1,2,3};
  int n = sizeof(arr)/sizeof(arr[0]);
  leader_array(arr,n);
}
