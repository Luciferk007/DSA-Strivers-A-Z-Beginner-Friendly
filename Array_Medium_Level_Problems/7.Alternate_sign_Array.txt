#include<iostream>
#include<bits/stdc++.h>
using namespace std;


void sign_unsign_arr(vector<int>arr)
{
  vector<int>positive;
  vector<int>negative;
  for(int i=0; i<arr.size(); i++)
  {
    if(arr[i] <0)
    {
      negative.push_back(arr[i]);
    }
    else
    {
      positive.push_back(arr[i]);
    }
  }
  //This interleaving only works if number of positive is equal to number of negatives
  for(int i=0; i<positive.size(); i++)
  {
    arr[2*i] = positive[i];
    arr[2*i + 1] = negative[i];
  }
  //To append no of unequal postive and negative values
  i=0; j=0;
  while(i<postive.size() && j<negative.size())
  {
    arr.push_back(postive[i++]);
    arr.push_back(negative[j++]);
  }

  {
    cout<<arr[i]<<" ";
  }
}

{
  vector<int>arr = {1,2,-3,-1,-2,-3};
