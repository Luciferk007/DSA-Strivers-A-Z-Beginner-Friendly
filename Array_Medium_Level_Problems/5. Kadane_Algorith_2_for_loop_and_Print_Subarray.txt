#include <iostream>
#include<bits/stdc++.h>
using namespace std;

//Brute force Approach
void kadane_algo_brute(int arr[], int n)
{
  int max_sum=INT_MIN;
  int index_start, index_end = 0;
  for(int i=0; i<n; i++)
  { 
    int sum =0;
    for(int j=i; j<n; j++)
    { 
      sum += arr[j];
      if(sum > max_sum)
      {
        max_sum = sum;
        index_start = i;
        index_end = j;
        cout<<"Index start:"<<i<<endl;
        cout<<"Index end: "<<j<<endl;
      }
    }
    
  }
  cout<<"MAX_SUM: "<<max_sum<<endl;
  for(int i=index_start; i<= index_end; i++)
  {
    cout<<i<<" ";
  }
}


int main()
{
  int arr[] = {-1,2,-3,4,5};
  int n = sizeof(arr)/sizeof(arr[0]);
  kadane_algo_brute(arr,n);
}
