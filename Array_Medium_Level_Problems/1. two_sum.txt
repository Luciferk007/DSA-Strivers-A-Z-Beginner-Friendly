#include<iostream>
#include<vector>
using namespace std;

void two_sum(int arr[],int target,int n)
{
  vector<int>result;
  for(int i=0; i<n; i++)
  {
    for(int j=i+1; j<n; j++)
    {
      if(arr[i] + arr[j] == target)
      {
        result.push_back(arr[i]);
        result.push_back(arr[j]);
        break; //To take only two elements breaking inner for loop
      }
    }
    break; //To only take two elements breaking outer for loop
  }
  for(int i=0; i<result.size(); i++)
  {
    cout<<result[i]<<endl;
  }
}



int main()
{
    int arr[] = {2, 6, 5, 8, 11};
    int n = 5;
    int target = 13;
    two_sum(arr, target, n);
}
