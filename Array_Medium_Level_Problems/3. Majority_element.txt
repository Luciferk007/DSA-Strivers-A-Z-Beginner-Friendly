#include<iostream>
#include<vector>
using namespace std;

void majority_elem(int arr[], int n)
{
  int max_count =0;
  for(int i=0; i<n; i++)
  { int count =0;
    for(int j=0; j<n; j++)
    {
      if(arr[i] == arr[j])
      {
        count++;
      }
    }
    if(count > n/2)
    {
      cout<<"Majority Element is:"<<arr[i]<<endl;
      break;
    }
  }
  
    
    
}


int main()
{
    int arr[] = {2, 3, 3, 3, 11};
    int n = 5;
    majority_elem(arr,n);
}
