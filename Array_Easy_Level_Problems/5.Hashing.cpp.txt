#include<iostream>
using namespace std;


int main()
{
  int q[] = {1,1,2,2,2,4,5,5,5,5};
  int n = sizeof(q)/sizeof(q[0]);
  int hash[6] = {0};
  int max_v=0;
  int min_v = 1000;
  for(int i=0; i<n; i++)
  {
    hash[q[i]] +=1;
  }
  int queries;
  cout<<"Pls enter the no of quries you wanna check\n"<<endl;
  cin>>queries;
  for(int i=0; i<queries; i++)
  {
    int number;
    cout<<"Enter the "<<queries<<" Numbers to check freq:"<<endl;
    cin>>number;
    cout<<"Freq of number : "<<number <<" is \n"<<hash[number]<<endl;
  }
  for(int i=0; i<n; i++)
  {
    cout<<"MAX freq number is: "<<max(max_v,hash[q[i]])<<endl;
    cout<<"Min freq number is: "<<min(min_v,hash[q[i]])<<endl;
  }
}
