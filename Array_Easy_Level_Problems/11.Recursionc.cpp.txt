int sumN(int i, int n)
{
  if(i<=n)
  { 
    return i+ sumN(i+1,n);
  }
  else
  {
    return 0;
  }
}

int factorialN(int i, int n)
{
  if(i<=n)
  {
    return i*factorialN(i+1,n);
  }
  else
  {
    return 1;
  }
}


void reverseN(int start, int end, int arr[])
{ 
  if(start<=end)
  {
    int temp;
    temp = arr[start];
    arr[start] = arr[end];
    arr[end] = temp;
    reverseN(start+1,end-1,arr);
  }
  
}

void reverseS(int start, int end, string &str)
{
  if(start<=end)
  {
    char temp = str[start];
    str[start] = str[end];
    str[end] = temp;
    reverseS(start+1, end-1, str);
  }
}

string reverseDirect(string s)
{
  reverse(s.begin(),s.end());
  return s;
}

void is_palindrome(string s)
{
  string temp = reverseDirect(s);
  int n = s.length();
  for(int i=0; i<n; i++)
  {
    if(s[i] == temp[i])
    { 
      cout<<s[i]<<"="<<temp[i]<<" Is a Palindrome"<<endl;
      continue;
    }
    else
    { 
      cout<<"Not a Palindrome"<<endl;
      break;
    }
  }
}
