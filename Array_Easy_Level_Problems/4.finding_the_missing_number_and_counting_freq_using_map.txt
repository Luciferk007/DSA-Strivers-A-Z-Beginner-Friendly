#include<iostream>
#include<map>
using namespace std;
void missing(int arr[], int n)
{
  map<int,int>m;
  //vector<int>vec;
  for(int i=0; i<n; i++)
  {
    m[arr[i]]++;
  }
  
  for(auto it : m)
  {
    if(it.second == 0)
    {
      cout<<"Missing no is:"<<it.first<<endl;
    }
  }
}

void missing_hash(int arr[], int n)
{
  int hash[n+1] = {0};
  for(int i=0; i<n; i++)
  {
    hash[arr[i]]++;
  }
  for(int i=1; i<n; i++)
  {
    if(hash[i] == 0)
    {
      cout<<"The missing number is:"<<i<<endl;
    }
  }
}

int main()
{
  int arr[] = {1,2,4,5};
  int n = sizeof(arr)/sizeof(arr[0]);
  //missing(arr,n);
  missing_hash(arr,n);
}
